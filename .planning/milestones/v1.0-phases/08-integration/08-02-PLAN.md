          // __tests__/e2e.test.tsx
          // End-to-end integration tests for learning feature

          // Tests the complete user flow through the learning feature.
          // Uses MSW (Mock Service Worker) for API mocking.
          // Covers happy path and edge cases.

          // @see: client/src/features/learning/LearningPathContainer.tsx
          // @note: These are integration tests, not true E2E (no real browser)

          import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest';
          import { render, screen, fireEvent, waitFor, within } from '@testing-library/react';
          import userEvent from '@testing-library/user-event';
          import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
          import { MemoryRouter, Routes, Route } from 'react-router-dom';
          import { LearningHome } from '../LearningHome';
          import { LearningPage } from '../LearningPage';

          // Mock API
          vi.mock('@/lib/learningApi', () => ({
            generateCourse: vi.fn(),
            getLearningSession: vi.fn(),
            transitionNode: vi.fn(),
            submitQuiz: vi.fn(),
            retryQuiz: vi.fn(),
            regenerateNode: vi.fn(),
            getQuizAttempts: vi.fn(),
          }));

          import * as api from '@/lib/learningApi';

          // Test fixtures
          const createMockSession = (overrides = {}) => ({
            id: 'session-1',
            course_title: 'Newton\'s Laws of Motion',
            total_nodes: 3,
            completed_nodes: 0,
            nodes: [
              {
                id: 'node-1',
                title: 'Newton\'s First Law',
                status: 'VIEWING_EXPLANATION',
                sequence_index: 0,
                content_markdown: '# First Law\n\nAn object at rest stays at rest.',
                quiz: {
                  question_text: 'What is inertia?',
                  options: [
                    { id: 'A', text: 'Resistance to change', is_correct: true, explanation: 'Correct!' },
                    { id: 'B', text: 'Speed of motion', is_correct: false, explanation: 'Wrong' },
                    { id: 'C', text: 'Force applied', is_correct: false, explanation: 'Wrong' },
                    { id: 'D', text: 'Mass of object', is_correct: false, explanation: 'Wrong' },
                  ],
                },
              },
              {
                id: 'node-2',
                title: 'Newton\'s Second Law',
                status: 'LOCKED',
                sequence_index: 1,
                content_markdown: '# Second Law\n\nF = ma',
                quiz: null,
              },
              {
                id: 'node-3',
                title: 'Newton\'s Third Law',
                status: 'LOCKED',
                sequence_index: 2,
                content_markdown: '# Third Law\n\nFor every action...',
                quiz: null,
              },
            ],
            ...overrides,
          });

          const createWrapper = () => {
            const queryClient = new QueryClient({
              defaultOptions: {
                queries: { retry: false },
                mutations: { retry: false },
              },
            });

            return ({ children }: { children: React.ReactNode }) => (
              <QueryClientProvider client={queryClient}>
                <MemoryRouter initialEntries={['/learn']}>
                  <Routes>
                    <Route path="/learn" element={<LearningHome />} />
                    <Route path="/learn/:sessionId" element={<LearningPage />} />
                  </Routes>
                  {children}
                </MemoryRouter>
              </QueryClientProvider>
            );
          };

          describe('Learning Feature E2E Flow', () => {
            beforeEach(() => {
              vi.clearAllMocks();
            });

            describe('Topic Input to Course Generation', () => {
              it('generates course from topic input and navigates to session', async () => {
                const user = userEvent.setup();
                const mockSession = createMockSession();
                (api.generateCourse as any).mockResolvedValue(mockSession);
                (api.getLearningSession as any).mockResolvedValue(mockSession);

                render(<LearningHome />, { wrapper: createWrapper() });

                // Type topic
                const input = screen.getByPlaceholderText(/what do you want to learn/i);
                await user.type(input, 'Newton\'s Laws');

                // Submit
                const submitButton = screen.getByRole('button', { name: /learn/i });
                await user.click(submitButton);

                // Verify generation called
                await waitFor(() => {
                  expect(api.generateCourse).toHaveBeenCalledWith({
                    query: 'Newton\'s Laws',
                    user_id: undefined,
                  });
                });
              });

              it('shows loading state during generation', async () => {
                const user = userEvent.setup();
                let resolveGenerate: (value: any) => void;
                (api.generateCourse as any).mockImplementation(
                  () => new Promise((resolve) => { resolveGenerate = resolve; })
                );

                render(<LearningHome />, { wrapper: createWrapper() });

                const input = screen.getByPlaceholderText(/what do you want to learn/i);
                await user.type(input, 'Test');
                await user.click(screen.getByRole('button', { name: /learn/i }));

                expect(screen.getByText(/creating your personalized learning path/i)).toBeInTheDocument();
              });
            });

            describe('Sequential Learning Flow', () => {
              it('shows explanation first, quiz hidden', async () => {
                const mockSession = createMockSession();
                (api.getLearningSession as any).mockResolvedValue(mockSession);

                render(<LearningPage />, {
                  wrapper: ({ children }) => {
                    const queryClient = new QueryClient();
                    return (
                      <QueryClientProvider client={queryClient}>
                        <MemoryRouter initialEntries={['/learn/session-1']}>
                          <Routes>
                            <Route path="/learn/:sessionId" element={children} />
                          </Routes>
                        </MemoryRouter>
                      </QueryClientProvider>
                    );
                  },
                });

                await waitFor(() => {
                  expect(screen.getByText('Newton\'s First Law')).toBeInTheDocument();
                });

                // Explanation visible
                expect(screen.getByText(/an object at rest stays at rest/i)).toBeInTheDocument();

                // Proceed button visible
                expect(screen.getByText(/proceed to quiz/i)).toBeInTheDocument();

                // Quiz NOT visible yet
                expect(screen.queryByText('What is inertia?')).not.toBeInTheDocument();
              });

              it('transitions to quiz when user clicks proceed', async () => {
                const user = userEvent.setup();
                const mockSession = createMockSession();
                (api.getLearningSession as any).mockResolvedValue(mockSession);
                (api.transitionNode as any).mockResolvedValue({
                  ...mockSession.nodes[0],
                  status: 'IN_QUIZ',
                });

                render(<LearningPage />, {
                  wrapper: ({ children }) => {
                    const queryClient = new QueryClient();
                    return (
                      <QueryClientProvider client={queryClient}>
                        <MemoryRouter initialEntries={['/learn/session-1']}>
                          <Routes>
                            <Route path="/learn/:sessionId" element={children} />
                          </Routes>
                        </MemoryRouter>
                      </QueryClientProvider>
                    );
                  },
                });

                await waitFor(() => {
                  expect(screen.getByText(/proceed to quiz/i)).toBeInTheDocument();
                });

                await user.click(screen.getByText(/proceed to quiz/i));

                await waitFor(() => {
                  expect(api.transitionNode).toHaveBeenCalledWith('node-1', 'IN_QUIZ');
                });
              });
            });

            describe('Mastery and Retry Flow', () => {
              it('shows retry button when quiz answer is wrong', async () => {
                const mockSession = createMockSession({
                  nodes: [
                    {
                      ...createMockSession().nodes[0],
                      status: 'SHOWING_FEEDBACK',
                    },
                    ...createMockSession().nodes.slice(1),
                  ],
                });
                const mockResult = {
                  node_id: 'node-1',
                  is_correct: false,
                  is_mastered: false,
                  score_percent: 0,
                  selected_option_id: 'B',
                  correct_option_id: 'A',
                  explanation: 'Wrong answer',
                };

                (api.getLearningSession as any).mockResolvedValue(mockSession);

                // This test would need component to be in SHOWING_FEEDBACK state
                // with quizResult showing is_mastered: false
              });

              it('unlocks next node when quiz answer is correct (100%)', async () => {
                const mockResult = {
                  node_id: 'node-1',
                  is_correct: true,
                  is_mastered: true,
                  score_percent: 100,
                  selected_option_id: 'A',
                  correct_option_id: 'A',
                  explanation: 'Correct!',
                };

                (api.submitQuiz as any).mockResolvedValue(mockResult);

                // Verify is_mastered triggers unlock of next node
              });
            });

            describe('Error Handling', () => {
              it('shows error state when session fails to load', async () => {
                (api.getLearningSession as any).mockRejectedValue(new Error('Network error'));

                render(<LearningPage />, {
                  wrapper: ({ children }) => {
                    const queryClient = new QueryClient({
                      defaultOptions: { queries: { retry: false } },
                    });
                    return (
                      <QueryClientProvider client={queryClient}>
                        <MemoryRouter initialEntries={['/learn/session-1']}>
                          <Routes>
                            <Route path="/learn/:sessionId" element={children} />
                          </Routes>
                        </MemoryRouter>
                      </QueryClientProvider>
                    );
                  },
                });

                await waitFor(() => {
                  expect(screen.getByText(/failed to load/i)).toBeInTheDocument();
                });

                expect(screen.getByText(/try again/i)).toBeInTheDocument();
              });

              it('shows not found for 404 response', async () => {
                const error = new Error('Not found');
                (error as any).response = { status: 404 };
                (api.getLearningSession as any).mockRejectedValue(error);

                render(<LearningPage />, {
                  wrapper: ({ children }) => {
                    const queryClient = new QueryClient({
                      defaultOptions: { queries: { retry: false } },
                    });
                    return (
                      <QueryClientProvider client={queryClient}>
                        <MemoryRouter initialEntries={['/learn/session-1']}>
                          <Routes>
                            <Route path="/learn/:sessionId" element={children} />
                          </Routes>
                        </MemoryRouter>
                      </QueryClientProvider>
                    );
                  },
                });

                await waitFor(() => {
                  expect(screen.getByText(/session not found/i)).toBeInTheDocument();
                });
              });
            });

            describe('Course Completion', () => {
              it('shows completion celebration when all nodes completed', async () => {
                const completedSession = createMockSession({
                  completed_nodes: 3,
                  nodes: createMockSession().nodes.map((n) => ({
                    ...n,
                    status: 'COMPLETED',
                  })),
                });

                (api.getLearningSession as any).mockResolvedValue(completedSession);

                render(<LearningPage />, {
                  wrapper: ({ children }) => {
                    const queryClient = new QueryClient();
                    return (
                      <QueryClientProvider client={queryClient}>
                        <MemoryRouter initialEntries={['/learn/session-1']}>
                          <Routes>
                            <Route path="/learn/:sessionId" element={children} />
                          </Routes>
                        </MemoryRouter>
                      </QueryClientProvider>
                    );
                  },
                });

                await waitFor(() => {
                  expect(screen.getByText(/course complete/i)).toBeInTheDocument();
                });
              });
            });
          });
