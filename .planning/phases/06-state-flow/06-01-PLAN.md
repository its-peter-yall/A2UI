          ```typescript
          // useNodeState.ts
          // Hook for determining available actions based on node status

          // Provides a declarative interface for node state and available actions.
          // Enforces sequential flow: explanation → quiz → feedback → (retry|complete)

          // @see: server/schemas/learning.py - NodeStatus enum definition
          // @note: Server is authoritative; this provides client-side UX guidance

          import type { NodeStatus, ConceptNode, QuizSubmitResponse } from '@/types/learning';

          interface NodeActions {
            canViewExplanation: boolean;
            canProceedToQuiz: boolean;
            canSubmitQuiz: boolean;
            canRetryQuiz: boolean;
            canContinueToNext: boolean;
            canRegenerate: boolean;
          }

          interface NodeStateResult {
            status: NodeStatus;
            actions: NodeActions;
            isLocked: boolean;
            isActive: boolean;
            isCompleted: boolean;
            isError: boolean;
            isMastered: boolean;
            currentView: 'locked' | 'explanation' | 'quiz' | 'feedback' | 'completed' | 'error';
          }

          export function useNodeState(
            node: ConceptNode,
            quizResult?: QuizSubmitResponse | null
          ): NodeStateResult {
            const { status } = node;

            // Determine current view based on status
            const currentView = getViewForStatus(status);

            // Check if mastered (from quiz result or status)
            const isMastered = quizResult?.is_mastered || status === 'COMPLETED';

            // Determine available actions based on sequential flow
            const actions: NodeActions = {
              // Can view explanation in VIEWING_EXPLANATION or COMPLETED (review)
              canViewExplanation:
                status === 'VIEWING_EXPLANATION' || status === 'COMPLETED',

              // Can proceed to quiz only from VIEWING_EXPLANATION
              canProceedToQuiz: status === 'VIEWING_EXPLANATION',

              // Can submit quiz only when IN_QUIZ
              canSubmitQuiz: status === 'IN_QUIZ',

              // Can retry only from SHOWING_FEEDBACK and not mastered
              canRetryQuiz: status === 'SHOWING_FEEDBACK' && !isMastered,

              // Can continue to next only from SHOWING_FEEDBACK and mastered
              canContinueToNext: status === 'SHOWING_FEEDBACK' && isMastered,

              // Can regenerate only from ERROR state
              canRegenerate: status === 'ERROR',
            };

            return {
              status,
              actions,
              isLocked: status === 'LOCKED',
              isActive: status !== 'LOCKED' && status !== 'COMPLETED' && status !== 'ERROR',
              isCompleted: status === 'COMPLETED',
              isError: status === 'ERROR',
              isMastered,
              currentView,
            };
          }

          function getViewForStatus(
            status: NodeStatus
          ): 'locked' | 'explanation' | 'quiz' | 'feedback' | 'completed' | 'error' {
            switch (status) {
              case 'LOCKED':
                return 'locked';
              case 'VIEWING_EXPLANATION':
                return 'explanation';
              case 'IN_QUIZ':
                return 'quiz';
              case 'SHOWING_FEEDBACK':
                return 'feedback';
              case 'COMPLETED':
                return 'completed';
              case 'ERROR':
                return 'error';
              default:
                return 'locked';
            }
          }

          /**
           * Check if a transition from current status to target is valid.
           * Mirrors server-side validation in learning_persistence.py
           */
          export function isValidTransition(
            current: NodeStatus,
            target: NodeStatus
          ): boolean {
            if (current === target) return true;

            const allowed: Record<NodeStatus, NodeStatus[]> = {
              LOCKED: ['VIEWING_EXPLANATION', 'ERROR'],
              VIEWING_EXPLANATION: ['IN_QUIZ', 'ERROR'],
              IN_QUIZ: ['SHOWING_FEEDBACK', 'ERROR'],
              SHOWING_FEEDBACK: ['IN_QUIZ', 'COMPLETED'], // Retry or master
              COMPLETED: [], // Terminal
              ERROR: ['LOCKED', 'VIEWING_EXPLANATION'],
            };

            return allowed[current]?.includes(target) ?? false;
          }
          ```

          ```typescript
          // useLearningMutations.ts
          // React Query mutations for sequential learning flow

          // Provides mutations for all learning state transitions with cache
          // invalidation, optimistic updates, and mastery-based progression.
          //
          // Sequential Flow:
          //   VIEWING_EXPLANATION → (proceedToQuiz) → IN_QUIZ
          //   IN_QUIZ → (submitQuiz) → SHOWING_FEEDBACK
          //   SHOWING_FEEDBACK → (retryQuiz) → IN_QUIZ [if not mastered]
          //   SHOWING_FEEDBACK → (continueToNext) → next node unlocks [if mastered]

          // @see: client/src/lib/learningApi.ts - API functions
          // @note: Mastery = 100% score required to unlock next topic

          import { useMutation, useQueryClient } from '@tanstack/react-query';
          import {
            transitionNode,
            submitQuiz,
            retryQuiz,
            regenerateNode,
          } from '@/lib/learningApi';
          import type {
            ConceptNode,
            NodeStatus,
            QuizSubmitResponse,
            LearningSessionWithNodes,
          } from '@/types/learning';
          import {
            optimisticStatusUpdate,
            optimisticCompletionUpdate,
            optimisticUnlockNext,
          } from './optimisticUpdates';

          interface UseLearningMutationsProps {
            sessionId: string;
            onQuizResult?: (result: QuizSubmitResponse) => void;
            onMasteryAchieved?: (nodeId: string) => void;
            onRetryNeeded?: (nodeId: string, result: QuizSubmitResponse) => void;
            onError?: (error: Error, context: string) => void;
          }

          export function useLearningMutations({
            sessionId,
            onQuizResult,
            onMasteryAchieved,
            onRetryNeeded,
            onError,
          }: UseLearningMutationsProps) {
            const queryClient = useQueryClient();

            // Helper to invalidate session data
            const invalidateSession = () => {
              queryClient.invalidateQueries({
                queryKey: ['learningSession', sessionId],
              });
            };

            // ============================================================
            // TRANSITION: VIEWING_EXPLANATION → IN_QUIZ
            // User clicks "I understand, proceed to quiz"
            // ============================================================
            const transitionMutation = useMutation({
              mutationFn: ({
                nodeId,
                targetStatus,
              }: {
                nodeId: string;
                targetStatus: NodeStatus;
              }) => transitionNode(nodeId, targetStatus),
              onMutate: async ({ nodeId, targetStatus }) => {
                await queryClient.cancelQueries({
                  queryKey: ['learningSession', sessionId],
                });

                const rollback = optimisticStatusUpdate(
                  queryClient,
                  sessionId,
                  nodeId,
                  targetStatus
                );

                return { rollback };
              },
              onError: (error, variables, context) => {
                context?.rollback?.();
                onError?.(error as Error, 'transition');
              },
              onSettled: () => {
                invalidateSession();
              },
            });

            // ============================================================
            // SUBMIT QUIZ: IN_QUIZ → SHOWING_FEEDBACK
            // User submits answer; server validates and returns result
            // ============================================================
            const submitQuizMutation = useMutation({
              mutationFn: ({
                nodeId,
                selectedOptionId,
              }: {
                nodeId: string;
                selectedOptionId: string;
              }) => submitQuiz(nodeId, { selected_option_id: selectedOptionId }),
              onMutate: async ({ nodeId }) => {
                await queryClient.cancelQueries({
                  queryKey: ['learningSession', sessionId],
                });

                // Optimistically transition to SHOWING_FEEDBACK
                const rollback = optimisticStatusUpdate(
                  queryClient,
                  sessionId,
                  nodeId,
                  'SHOWING_FEEDBACK'
                );

                return { rollback };
              },
              onSuccess: (result, { nodeId }) => {
                // Always call the generic quiz result callback
                onQuizResult?.(result);

                // Handle mastery vs retry paths
                if (result.is_mastered) {
                  // User achieved 100% - can proceed to next
                  onMasteryAchieved?.(nodeId);

                  // Optimistically update: mark as completed, increment count, unlock next
                  const session = queryClient.getQueryData<LearningSessionWithNodes>([
                    'learningSession',
                    sessionId,
                  ]);
                  const nodeIndex = session?.nodes.findIndex((n) => n.id === nodeId) ?? -1;

                  if (nodeIndex >= 0) {
                    optimisticStatusUpdate(queryClient, sessionId, nodeId, 'COMPLETED');
                    optimisticCompletionUpdate(queryClient, sessionId, 1);
                    optimisticUnlockNext(queryClient, sessionId, nodeIndex);
                  }
                } else {
                  // User did not achieve 100% - must retry
                  onRetryNeeded?.(nodeId, result);
                }
              },
              onError: (error, variables, context) => {
                context?.rollback?.();
                onError?.(error as Error, 'submitQuiz');
              },
              onSettled: () => {
                invalidateSession();
              },
            });

            // ============================================================
            // RETRY QUIZ: SHOWING_FEEDBACK → IN_QUIZ
            // User clicks "Retry" after incorrect answer
            // ============================================================
            const retryQuizMutation = useMutation({
              mutationFn: (nodeId: string) => retryQuiz(nodeId),
              onMutate: async (nodeId) => {
                await queryClient.cancelQueries({
                  queryKey: ['learningSession', sessionId],
                });

                // Optimistically transition back to IN_QUIZ
                const rollback = optimisticStatusUpdate(
                  queryClient,
                  sessionId,
                  nodeId,
                  'IN_QUIZ'
                );

                return { rollback };
              },
              onError: (error, nodeId, context) => {
                context?.rollback?.();
                onError?.(error as Error, 'retryQuiz');
              },
              onSettled: () => {
                invalidateSession();
              },
            });

            // ============================================================
            // REGENERATE: ERROR → VIEWING_EXPLANATION
            // User clicks "Retry" on a failed generation
            // ============================================================
            const regenerateMutation = useMutation({
              mutationFn: (nodeId: string) => regenerateNode(nodeId),
              onSuccess: () => {
                invalidateSession();
              },
              onError: (error: Error) => {
                onError?.(error, 'regenerate');
              },
            });

            // ============================================================
            // Convenience functions for cleaner component code
            // ============================================================

            /**
             * Proceed from explanation to quiz.
             * Only valid from VIEWING_EXPLANATION state.
             */
            const proceedToQuiz = (nodeId: string) => {
              transitionMutation.mutate({
                nodeId,
                targetStatus: 'IN_QUIZ',
              });
            };

            /**
             * Submit quiz answer.
             * Only valid from IN_QUIZ state.
             * Result determines if user can proceed or must retry.
             */
            const submitAnswer = (nodeId: string, optionId: string) => {
              submitQuizMutation.mutate({
                nodeId,
                selectedOptionId: optionId,
              });
            };

            /**
             * Retry quiz after incorrect answer.
             * Only valid from SHOWING_FEEDBACK state when not mastered.
             */
            const retry = (nodeId: string) => {
              retryQuizMutation.mutate(nodeId);
            };

            /**
             * Continue to next topic after mastery.
             * Triggers scroll to next node and marks current as reviewed.
             * Only valid from SHOWING_FEEDBACK state when mastered.
             */
            const continueToNext = (nodeId: string, nextNodeId?: string) => {
              // Status is already COMPLETED from submitAnswer success path
              // Just need to scroll to next node
              if (nextNodeId) {
                document.getElementById(`node-${nextNodeId}`)?.scrollIntoView({
                  behavior: 'smooth',
                  block: 'center',
                });
              }
            };

            /**
             * Regenerate a failed node.
             * Only valid from ERROR state.
             */
            const regenerate = (nodeId: string) => {
              regenerateMutation.mutate(nodeId);
            };

            return {
              // Raw mutations (for advanced use cases)
              transitionMutation,
              submitQuizMutation,
              retryQuizMutation,
              regenerateMutation,

              // Convenience functions (recommended for most use cases)
              proceedToQuiz,
              submitAnswer,
              retry,
              continueToNext,
              regenerate,

              // Loading states
              isTransitioning: transitionMutation.isPending,
              isSubmitting: submitQuizMutation.isPending,
              isRetrying: retryQuizMutation.isPending,
              isRegenerating: regenerateMutation.isPending,
              isAnyLoading:
                transitionMutation.isPending ||
                submitQuizMutation.isPending ||
                retryQuizMutation.isPending ||
                regenerateMutation.isPending,
            };
          }
          ```
        </change>
      </changes>

          ```typescript
          // optimisticUpdates.ts
          // Optimistic update utilities for learning mutations

          // Provides functions to optimistically update the React Query cache
          // before mutations complete, with rollback support on error.
          // Supports the sequential flow state machine.

          // @see: client/src/types/learning.ts - Type definitions
          // @note: Always call invalidateQueries after mutation settles

          import type { QueryClient } from '@tanstack/react-query';
          import type {
            LearningSessionWithNodes,
            NodeStatus,
          } from '@/types/learning';

          /**
           * Optimistically update a node's status in the cache.
           * Returns a rollback function to restore previous state on error.
           */
          export function optimisticStatusUpdate(
            queryClient: QueryClient,
            sessionId: string,
            nodeId: string,
            newStatus: NodeStatus
          ): () => void {
            const previousData = queryClient.getQueryData<LearningSessionWithNodes>([
              'learningSession',
              sessionId,
            ]);

            if (!previousData) {
              return () => {}; // No-op rollback
            }

            queryClient.setQueryData<LearningSessionWithNodes>(
              ['learningSession', sessionId],
              {
                ...previousData,
                nodes: previousData.nodes.map((node) =>
                  node.id === nodeId ? { ...node, status: newStatus } : node
                ),
              }
            );

            // Return rollback function
            return () => {
              queryClient.setQueryData(['learningSession', sessionId], previousData);
            };
          }

          /**
           * Optimistically update completed_nodes count.
           */
          export function optimisticCompletionUpdate(
            queryClient: QueryClient,
            sessionId: string,
            increment: number
          ): () => void {
            const previousData = queryClient.getQueryData<LearningSessionWithNodes>([
              'learningSession',
              sessionId,
            ]);

            if (!previousData) {
              return () => {};
            }

            queryClient.setQueryData<LearningSessionWithNodes>(
              ['learningSession', sessionId],
              {
                ...previousData,
                completed_nodes: previousData.completed_nodes + increment,
              }
            );

            return () => {
              queryClient.setQueryData(['learningSession', sessionId], previousData);
            };
          }

          /**
           * Optimistically unlock the next node after mastery.
           * Sets next node status to VIEWING_EXPLANATION (not LOCKED).
           */
          export function optimisticUnlockNext(
            queryClient: QueryClient,
            sessionId: string,
            currentNodeIndex: number
          ): () => void {
            const previousData = queryClient.getQueryData<LearningSessionWithNodes>([
              'learningSession',
              sessionId,
            ]);

            if (!previousData) {
              return () => {};
            }

            const nextIndex = currentNodeIndex + 1;
            if (nextIndex >= previousData.nodes.length) {
              return () => {}; // No next node - course completed
            }

            // Only unlock if next node is currently LOCKED
            const nextNode = previousData.nodes[nextIndex];
            if (nextNode.status !== 'LOCKED') {
              return () => {}; // Already unlocked somehow
            }

            queryClient.setQueryData<LearningSessionWithNodes>(
              ['learningSession', sessionId],
              {
                ...previousData,
                nodes: previousData.nodes.map((node, idx) =>
                  idx === nextIndex
                    ? { ...node, status: 'VIEWING_EXPLANATION' as NodeStatus }
                    : node
                ),
              }
            );

            return () => {
              queryClient.setQueryData(['learningSession', sessionId], previousData);
            };
          }

          /**
           * Batch optimistic update for mastery: complete current + unlock next.
           * Combines multiple updates into a single cache write for efficiency.
           */
          export function optimisticMasteryUpdate(
            queryClient: QueryClient,
            sessionId: string,
            nodeId: string,
            nodeIndex: number
          ): () => void {
            const previousData = queryClient.getQueryData<LearningSessionWithNodes>([
              'learningSession',
              sessionId,
            ]);

            if (!previousData) {
              return () => {};
            }

            const nextIndex = nodeIndex + 1;
            const hasNextNode = nextIndex < previousData.nodes.length;

            queryClient.setQueryData<LearningSessionWithNodes>(
              ['learningSession', sessionId],
              {
                ...previousData,
                completed_nodes: previousData.completed_nodes + 1,
                nodes: previousData.nodes.map((node, idx) => {
                  if (node.id === nodeId) {
                    return { ...node, status: 'COMPLETED' as NodeStatus };
                  }
                  if (hasNextNode && idx === nextIndex && node.status === 'LOCKED') {
                    return { ...node, status: 'VIEWING_EXPLANATION' as NodeStatus };
                  }
                  return node;
                }),
              }
            );

            return () => {
              queryClient.setQueryData(['learningSession', sessionId], previousData);
            };
          }
          ```
